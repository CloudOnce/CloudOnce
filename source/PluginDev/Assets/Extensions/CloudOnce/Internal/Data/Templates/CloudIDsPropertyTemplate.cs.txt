            public static string INTERNALID
            {
                get
                {
#if UNITY_ANDROID && !UNITY_EDITOR
#if CLOUDONCE_AMAZON
                    return "AMAZONID";
#elif CLOUDONCE_GOOGLE
                    return "GOOGLEID";
#else
                    return string.Empty;
#endif
#elif UNITY_IOS && !UNITY_EDITOR
                    return "APPLEID";
#elif UNITY_EDITOR
                    return "INTERNALID";
#else
                    return string.Empty;
#endif
                }
            }
