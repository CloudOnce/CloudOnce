            public static string INTERNALID
            {
                get
                {
#if UNITY_ANDROID && !UNITY_EDITOR
#if CLOUDONCE_GOOGLE
                    return "GOOGLEID";
#else
                    return string.Empty;
#endif
#elif (UNITY_IOS || UNITY_TVOS) && !UNITY_EDITOR
                    return "APPLEID";
#elif UNITY_EDITOR
                    return "INTERNALID";
#else
                    return string.Empty;
#endif
                }
            }
